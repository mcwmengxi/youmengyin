    }

    width: 8in;
  }

  color: red;

bem ​

项目使用了element-plus,默认的混入不能冲突

css
@mixin B($block){
  $B:#{$Namespace + $block-sel + $block};
  .#{$B}{
    // 占位符 slot
    @content;
  }
}
// <input class="el-input__inner" />
@mixin E($el){
  $selector: &;
  @at-root{
    #{$selector + $element-sel + $el}{
      @content;
    }
  }
}
@mixin M($m){
  $selector: &;
  @at-root{
    #{$selector + $modify-sel + $m}{
      @content;
    }
  }
}

/* vite.config.ts导入 */
css: {
  preprocessorOptions: {
    scss: {
      // 在组件中使用SCSS全局变量
      additionalData: `@use "@/styles/elementplus/index.scss" as *;@use "@/styles/variables.scss" as *;@use "@/styles/bem.scss" as *;`
    }
  }
},

/* 使用 */
@include B(test) {
	text-shadow: 1ch;
	@include E(wrapper) {
		background: plum;
		@include M(primary) {
			color: greenyellow;
		}
	}
}
